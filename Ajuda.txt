# 1. Crie o projeto novo
composer create-project laravel/laravel despacho-novo

cd despacho-novo

# 2. Instale os pacotes essenciais
composer require laravel/sanctum livewire/livewire bacon/bacon-qr-code dompdf/dompdf phpmailer/phpmailer

# Frontend bonito
php artisan vendor:publish --provider="Livewire\LivewireServiceProvider"
npm install -D tailwindcss @tailwindcss/forms alpinejs daisyui flowbite
npx tailwindcss init -p

# 3. Auth rápido e bonito
composer require laravel/breeze --dev
php artisan breeze:install blade --dark
npm install && npm run build

# 4. Migrações (vou criar as tabelas de despacho, rastreamento, usuários, etc.)
# Eu crio tudo: users, envios, rastreamentos, logs, etc.

# 5. Rodar
php artisan migrate
php artisan serve
